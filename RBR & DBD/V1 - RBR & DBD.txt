//@version=6
indicator("RBR / DBD Zone - Trend Based Flexible Signal TP 5 Point", overlay=true, max_boxes_count=200, max_labels_count=500)

// =====================
// INPUTS
// =====================
showRBR = input.bool(true, "Show RBR Demand")
showDBD = input.bool(true, "Show DBD Supply")
showRetestSignals = input.bool(true, "Show BUY/SELL Signals")

showPatternLetters = input.bool(true, "Show RBR / DBD Candle Letters")
patternLabelSize = input.string("normal", "Pattern Letter Size", options=["tiny", "small", "normal", "large"])
patternOffsetAtr = input.float(0.30, "Pattern Letter Offset ATR", step=0.05)

zoneType = input.string("Wick", "Zone Type", options=["Wick", "Body", "Hybrid"])
hybridBufferAtr = input.float(0.10, "Hybrid Buffer ATR", step=0.05)

minBaseCandles = input.int(1, "Min Base Candles", minval=1, maxval=10)
maxBaseCandles = input.int(6, "Max Base Candles", minval=2, maxval=15)

atrLen = input.int(14, "ATR Length")
impulseAtrMult = input.float(0.80, "Impulse ATR Multiplier", step=0.05)
impulseBodyPct = input.float(0.45, "Min Impulse Body %", step=0.05)
baseAtrMult = input.float(1.00, "Max Base Candle ATR Multiplier", step=0.05)
baseBodyPct = input.float(0.70, "Max Base Body %", step=0.05)
baseTotalAtrMult = input.float(2.20, "Max Total Base Size ATR", step=0.1)

zoneBoxLength = input.int(20, "Zone Box Length", minval=1, maxval=100)
signalOnlyWhenBoxVisible = input.bool(true, "Signal Only When Box Is Visible")
maxZones = input.int(30, "Max Zones", minval=5, maxval=100)

confirmMode = input.string("Close Direction", "Retest Confirmation", options=["None", "Close Direction", "Wick Rejection"])
wickMult = input.float(0.60, "Wick Rejection Multiplier", step=0.1)

useTrendFilter = input.bool(true, "Use EMA Trend Filter For Box And Signal")
emaLen = input.int(50, "EMA Length")

strictRetest = input.bool(true, "Strict Retest")
retestCloseMode = input.string("Mid Zone", "Retest Close Requirement", options=["Outside Zone", "Mid Zone", "Inside Zone OK"])
allowBodyInsideZone = input.bool(true, "Allow Body To Enter Zone")

tpPoints = input.float(5.0, "TP Points", step=0.1)
showTPLabel = input.bool(true, "Show TP In Label")

enableAlertFunction = input.bool(true, "Enable Any alert() Function Call")
alertOnBarCloseOnly = input.bool(true, "Alert Only After Candle Close")

// =====================
// CALC
// =====================
atr = ta.atr(atrLen)
ema = ta.ema(close, emaLen)

plot(useTrendFilter ? ema : na, "EMA", color=color.orange)

letterSize = patternLabelSize == "tiny" ? size.tiny : patternLabelSize == "small" ? size.small : patternLabelSize == "large" ? size.large : size.normal

bodySize(offset) =>
    math.abs(close[offset] - open[offset])

rangeSize(offset) =>
    high[offset] - low[offset]

bodyPct(offset) =>
    r = rangeSize(offset)
    r > 0 ? bodySize(offset) / r : 0.0

isBullImpulse(offset) =>
    close[offset] > open[offset] and rangeSize(offset) >= atr[offset] * impulseAtrMult and bodyPct(offset) >= impulseBodyPct

isBearImpulse(offset) =>
    close[offset] < open[offset] and rangeSize(offset) >= atr[offset] * impulseAtrMult and bodyPct(offset) >= impulseBodyPct

isBaseCandle(offset) =>
    rangeSize(offset) <= atr[offset] * baseAtrMult and bodyPct(offset) <= baseBodyPct

trendBuyOK = not useTrendFilter or close > ema
trendSellOK = not useTrendFilter or close < ema

// =====================
// ARRAYS
// =====================
var array<box> zoneBoxes = array.new_box()
var array<float> zoneTops = array.new_float()
var array<float> zoneBottoms = array.new_float()
var array<int> zoneTypes = array.new_int()
var array<bool> zoneUsed = array.new_bool()
var array<int> zoneBornBar = array.new_int()
var array<int> zoneEndBar = array.new_int()

var int lastRBRBar = na
var int lastDBDBar = na

// =====================
// REMOVE ZONE FUNCTION
// =====================
removeZone(_i) =>
    bx = array.get(zoneBoxes, _i)
    box.delete(bx)
    array.remove(zoneBoxes, _i)
    array.remove(zoneTops, _i)
    array.remove(zoneBottoms, _i)
    array.remove(zoneTypes, _i)
    array.remove(zoneUsed, _i)
    array.remove(zoneBornBar, _i)
    array.remove(zoneEndBar, _i)

// =====================
// MARK PATTERN LETTERS
// =====================
markRBR(_len) =>
    if showPatternLetters
        float offset = atr * patternOffsetAtr

        label.new(
             bar_index - _len - 1,
             high[_len + 1] + offset,
             "R",
             style=label.style_none,
             textcolor=color.lime,
             size=letterSize
         )

        for k = 1 to _len
            label.new(
                 bar_index - k,
                 high[k] + offset,
                 "B",
                 style=label.style_none,
                 textcolor=color.yellow,
                 size=letterSize
             )

        label.new(
             bar_index,
             high + offset,
             "R",
             style=label.style_none,
             textcolor=color.lime,
             size=letterSize
         )

markDBD(_len) =>
    if showPatternLetters
        float offset = atr * patternOffsetAtr

        label.new(
             bar_index - _len - 1,
             low[_len + 1] - offset,
             "D",
             style=label.style_none,
             textcolor=color.red,
             size=letterSize
         )

        for k = 1 to _len
            label.new(
                 bar_index - k,
                 low[k] - offset,
                 "B",
                 style=label.style_none,
                 textcolor=color.yellow,
                 size=letterSize
             )

        label.new(
             bar_index,
             low - offset,
             "D",
             style=label.style_none,
             textcolor=color.red,
             size=letterSize
         )

// =====================
// DETECT NEW ZONE
// =====================
bool newRBR = false
bool newDBD = false

float rbrTop = na
float rbrBot = na
int rbrLen = na

float dbdTop = na
float dbdBot = na
int dbdLen = na

for len = minBaseCandles to maxBaseCandles
    bool baseOK = true

    float wickTop = high[1]
    float wickBot = low[1]
    float bodyTop = math.max(open[1], close[1])
    float bodyBot = math.min(open[1], close[1])

    for j = 1 to len
        baseOK := baseOK and isBaseCandle(j)
        wickTop := math.max(wickTop, high[j])
        wickBot := math.min(wickBot, low[j])
        bodyTop := math.max(bodyTop, math.max(open[j], close[j]))
        bodyBot := math.min(bodyBot, math.min(open[j], close[j]))

    float zoneTop = zoneType == "Wick" ? wickTop : zoneType == "Body" ? bodyTop : bodyTop + atr * hybridBufferAtr
    float zoneBot = zoneType == "Wick" ? wickBot : zoneType == "Body" ? bodyBot : bodyBot - atr * hybridBufferAtr

    bool baseTightOK = (wickTop - wickBot) <= atr * baseTotalAtrMult

    bool rbrOK = showRBR and trendBuyOK and baseOK and baseTightOK and isBullImpulse(len + 1) and isBullImpulse(0) and close > wickTop
    bool dbdOK = showDBD and trendSellOK and baseOK and baseTightOK and isBearImpulse(len + 1) and isBearImpulse(0) and close < wickBot

    if rbrOK and not newRBR and (na(lastRBRBar) or bar_index > lastRBRBar + len)
        newRBR := true
        rbrTop := zoneTop
        rbrBot := zoneBot
        rbrLen := len

    if dbdOK and not newDBD and (na(lastDBDBar) or bar_index > lastDBDBar + len)
        newDBD := true
        dbdTop := zoneTop
        dbdBot := zoneBot
        dbdLen := len

if newRBR and newDBD
    if close >= open
        newDBD := false
    else
        newRBR := false

// =====================
// CREATE RBR DEMAND
// =====================
if newRBR
    lastRBRBar := bar_index
    int endBar = bar_index + zoneBoxLength

    b = box.new(
         left=bar_index - rbrLen,
         top=rbrTop,
         right=endBar,
         bottom=rbrBot,
         bgcolor=color.new(color.lime, 85),
         border_color=color.new(color.lime, 15),
         text="RBR Demand",
         text_color=color.lime,
         text_size=size.tiny
     )

    array.unshift(zoneBoxes, b)
    array.unshift(zoneTops, rbrTop)
    array.unshift(zoneBottoms, rbrBot)
    array.unshift(zoneTypes, 1)
    array.unshift(zoneUsed, false)
    array.unshift(zoneBornBar, bar_index)
    array.unshift(zoneEndBar, endBar)

    markRBR(rbrLen)

// =====================
// CREATE DBD SUPPLY
// =====================
if newDBD
    lastDBDBar := bar_index
    int endBar = bar_index + zoneBoxLength

    b = box.new(
         left=bar_index - dbdLen,
         top=dbdTop,
         right=endBar,
         bottom=dbdBot,
         bgcolor=color.new(color.red, 85),
         border_color=color.new(color.red, 15),
         text="DBD Supply",
         text_color=color.red,
         text_size=size.tiny
     )

    array.unshift(zoneBoxes, b)
    array.unshift(zoneTops, dbdTop)
    array.unshift(zoneBottoms, dbdBot)
    array.unshift(zoneTypes, -1)
    array.unshift(zoneUsed, false)
    array.unshift(zoneBornBar, bar_index)
    array.unshift(zoneEndBar, endBar)

    markDBD(dbdLen)

// =====================
// LIMIT OLD ZONES
// =====================
while array.size(zoneBoxes) > maxZones
    oldBox = array.pop(zoneBoxes)
    box.delete(oldBox)
    array.pop(zoneTops)
    array.pop(zoneBottoms)
    array.pop(zoneTypes)
    array.pop(zoneUsed)
    array.pop(zoneBornBar)
    array.pop(zoneEndBar)

// =====================
// REMOVE INVALID ZONES
// =====================
if array.size(zoneBoxes) > 0
    int szRemove = array.size(zoneBoxes)

    for x = 0 to szRemove - 1
        if x >= array.size(zoneBoxes)
            break

        i = array.size(zoneBoxes) - 1 - x

        if i < 0
            break

        top = array.get(zoneTops, i)
        bot = array.get(zoneBottoms, i)
        typ = array.get(zoneTypes, i)

        bool demandInvalid = typ == 1 and close < bot
        bool supplyInvalid = typ == -1 and close > top

        if demandInvalid or supplyInvalid
            removeZone(i)

// =====================
// FLEXIBLE RETEST ZONE CHECK
// =====================
int rbrIndex = na
int dbdIndex = na

if array.size(zoneBoxes) > 0
    for i = 0 to array.size(zoneBoxes) - 1
        top = array.get(zoneTops, i)
        bot = array.get(zoneBottoms, i)
        typ = array.get(zoneTypes, i)
        used = array.get(zoneUsed, i)
        born = array.get(zoneBornBar, i)
        endBar = array.get(zoneEndBar, i)

        float mid = (top + bot) / 2.0

        bool boxVisibleNow = bar_index >= born and bar_index <= endBar
        bool canSignalByBox = not signalOnlyWhenBoxVisible or boxVisibleNow
        bool afterCreated = bar_index > born + 1

        bool touchDemandZone = low <= top and high >= bot
        bool touchSupplyZone = high >= bot and low <= top

        bool buyCloseRequirement = retestCloseMode == "Outside Zone" ? close > top : retestCloseMode == "Mid Zone" ? close > mid : close >= bot
        bool sellCloseRequirement = retestCloseMode == "Outside Zone" ? close < bot : retestCloseMode == "Mid Zone" ? close < mid : close <= top

        bool buyBodyAllowed = allowBodyInsideZone or math.min(open, close) >= top
        bool sellBodyAllowed = allowBodyInsideZone or math.max(open, close) <= bot

        bool rbrCanSignal = canSignalByBox and afterCreated and not used and typ == 1 and touchDemandZone and buyBodyAllowed and (not strictRetest or buyCloseRequirement)
        bool dbdCanSignal = canSignalByBox and afterCreated and not used and typ == -1 and touchSupplyZone and sellBodyAllowed and (not strictRetest or sellCloseRequirement)

        if rbrCanSignal and na(rbrIndex)
            rbrIndex := i

        if dbdCanSignal and na(dbdIndex)
            dbdIndex := i

// =====================
// SIGNALS
// =====================
bool buySignal = false
bool sellSignal = false

float candleBody = math.abs(close - open)
float lowerWick = math.min(open, close) - low
float upperWick = high - math.max(open, close)

bool onlyInRBR = not na(rbrIndex) and na(dbdIndex)
bool onlyInDBD = not na(dbdIndex) and na(rbrIndex)

bool allowAlertThisBar = alertOnBarCloseOnly ? barstate.isconfirmed : true

float buyTP = na
float sellTP = na

// =====================
// BUY SIGNAL
// =====================
if showRetestSignals and onlyInRBR
    bool buyConfirm = trendBuyOK

    if confirmMode == "Close Direction"
        buyConfirm := buyConfirm and close > open

    if confirmMode == "Wick Rejection"
        buyConfirm := buyConfirm and close > open and lowerWick >= candleBody * wickMult

    if buyConfirm
        buySignal := true
        buyTP := close + tpPoints
        array.set(zoneUsed, rbrIndex, true)

        string buyLabelText = showTPLabel ? "BUY\nEntry: " + str.tostring(close, format.mintick) + "\nTP: " + str.tostring(buyTP, format.mintick) : "BUY"

        label.new(
             bar_index,
             low,
             buyLabelText,
             style=label.style_label_up,
             textcolor=color.white,
             color=color.new(color.green, 0),
             size=size.small
         )

// =====================
// SELL SIGNAL
// =====================
if showRetestSignals and onlyInDBD
    bool sellConfirm = trendSellOK

    if confirmMode == "Close Direction"
        sellConfirm := sellConfirm and close < open

    if confirmMode == "Wick Rejection"
        sellConfirm := sellConfirm and close < open and upperWick >= candleBody * wickMult

    if sellConfirm
        sellSignal := true
        sellTP := close - tpPoints
        array.set(zoneUsed, dbdIndex, true)

        string sellLabelText = showTPLabel ? "SELL\nEntry: " + str.tostring(close, format.mintick) + "\nTP: " + str.tostring(sellTP, format.mintick) : "SELL"

        label.new(
             bar_index,
             high,
             sellLabelText,
             style=label.style_label_down,
             textcolor=color.white,
             color=color.new(color.red, 0),
             size=size.small
         )

// =====================
// ALERT FUNCTION CALLS
// =====================
if enableAlertFunction and allowAlertThisBar and buySignal
    alert(
         "BUY | RBR Demand" +
         " | Symbol: " + syminfo.ticker +
         " | TF: " + timeframe.period +
         " | Entry: " + str.tostring(close, format.mintick) +
         " | TP " + str.tostring(tpPoints) + " Point: " + str.tostring(buyTP, format.mintick),
         alert.freq_once_per_bar_close
     )

if enableAlertFunction and allowAlertThisBar and sellSignal
    alert(
         "SELL | DBD Supply" +
         " | Symbol: " + syminfo.ticker +
         " | TF: " + timeframe.period +
         " | Entry: " + str.tostring(close, format.mintick) +
         " | TP " + str.tostring(tpPoints) + " Point: " + str.tostring(sellTP, format.mintick),
         alert.freq_once_per_bar_close
     )

// =====================
// PLOTS + STANDARD ALERTCONDITION
// =====================
plotshape(buySignal, title="BUY Signal", style=shape.triangleup, location=location.belowbar, color=color.lime, size=size.small, text="BUY")
plotshape(sellSignal, title="SELL Signal", style=shape.triangledown, location=location.abovebar, color=color.red, size=size.small, text="SELL")

alertcondition(buySignal, title="BUY RBR Retest", message="BUY setup: candle retest RBR Demand zone")
alertcondition(sellSignal, title="SELL DBD Retest", message="SELL setup: candle retest DBD Supply zone")
