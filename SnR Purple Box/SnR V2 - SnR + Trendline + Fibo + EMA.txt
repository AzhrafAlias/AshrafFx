//=============================
// TRENDLINE PRICE FUNCTIONS
//=============================
f_line_price_now(_line) =>
    x1 = line.get_x1(_line)
    y1 = line.get_y1(_line)
    x2 = line.get_x2(_line)
    y2 = line.get_y2(_line)
    dx = x2 - x1
    dx == 0 ? y2 : y1 + (y2 - y1) * (time - x1) / dx

f_line_price_prev(_line) =>
    x1 = line.get_x1(_line)
    y1 = line.get_y1(_line)
    x2 = line.get_x2(_line)
    y2 = line.get_y2(_line)
    dx = x2 - x1
    dx == 0 ? y2 : y1 + (y2 - y1) * (time[1] - x1) / dx

//=============================
// HELPER FUNCTIONS
//=============================
f_breaks_above(_level) =>
    not na(_level) and close > (_level + breakBuffer) and close[1] <= (_level + breakBuffer)

f_breaks_below(_level) =>
    not na(_level) and close < (_level - breakBuffer) and close[1] >= (_level - breakBuffer)

f_near_level(_price, _level, _buf) =>
    not na(_price) and not na(_level) and math.abs(_price - _level) <= _buf

f_bull_retest_in_zone_strict(_top, _bot) =>
    touchedInside = low <= _top and low >= _bot
    bullishBody = close > open
    closeAboveBot = close >= _bot
    lowerWick = math.min(open, close) - low
    bodySize = math.abs(close - open)
    wickOk = lowerWick >= bodySize * 0.2
    touchedInside and bullishBody and closeAboveBot and wickOk

f_bear_retest_in_zone_strict(_top, _bot) =>
    touchedInside = high >= _bot and high <= _top
    bearishBody = close < open
    closeBelowTop = close <= _top
    upperWick = high - math.max(open, close)
    bodySize = math.abs(close - open)
    wickOk = upperWick >= bodySize * 0.2
    touchedInside and bearishBody and closeBelowTop and wickOk

f_bull_retest_ok(_level) =>
    touched = low <= _level
    heldAbove = close >= _level
    bullishBody = close > open
    lowerWick = math.min(open, close) - low
    bodySize = math.abs(close - open)
    wickOk = lowerWick >= bodySize * 0.2
    touched and heldAbove and bullishBody and wickOk

f_bear_retest_ok(_level) =>
    touched = high >= _level
    heldBelow = close <= _level
    bearishBody = close < open
    upperWick = high - math.max(open, close)
    bodySize = math.abs(close - open)
    wickOk = upperWick >= bodySize * 0.2
    touched and heldBelow and bearishBody and wickOk
