//=========================
// M5 CONFIRM FUNCTIONS
//=========================
f_m5BullConfirm(float o, float h, float l, float c, float po, float phh, float pll, float pc, float lineOrTop, float bottom, bool isZone) =>
    body = math.abs(c - o)
    lowerWick = math.min(o, c) - l

    prevBodyHigh = math.max(po, pc)
    prevBodyLow = math.min(po, pc)
    currBodyHigh = math.max(o, c)
    currBodyLow = math.min(o, c)

    zoneMid = (lineOrTop + bottom) / 2.0

    touchArea = isZone ?
         (l <= lineOrTop + zoneTol and h >= bottom - zoneTol) :
         (l <= lineOrTop + lineTol and h >= lineOrTop - lineTol)

    closeOK = isZone ? c > zoneMid : c > lineOrTop
    bull = c > o

    rejection =
         touchArea and
         bull and
         closeOK and
         lowerWick >= body * wickRejectMult

    strictEngulf =
         bull and
         pc < po and
         currBodyLow <= prevBodyLow and
         currBodyHigh >= prevBodyHigh

    flexEngulf =
         bull and
         c > prevBodyHigh

    engulf =
         touchArea and
         closeOK and
         (allowFlexibleEngulfing ? (strictEngulf or flexEngulf) : strictEngulf)

    confirmMode == "Rejection Only" ? rejection :
     confirmMode == "Engulfing Only" ? engulf :
     rejection or engulf

f_m5BearConfirm(float o, float h, float l, float c, float po, float phh, float pll, float pc, float lineOrBottom, float top, bool isZone) =>
    body = math.abs(c - o)
    upperWick = h - math.max(o, c)

    prevBodyHigh = math.max(po, pc)
    prevBodyLow = math.min(po, pc)
    currBodyHigh = math.max(o, c)
    currBodyLow = math.min(o, c)

    zoneMid = (top + lineOrBottom) / 2.0

    touchArea = isZone ?
         (h >= lineOrBottom - zoneTol and l <= top + zoneTol) :
         (h >= lineOrBottom - lineTol and l <= lineOrBottom + lineTol)

    closeOK = isZone ? c < zoneMid : c < lineOrBottom
    bear = c < o

    rejection =
         touchArea and
         bear and
         closeOK and
         upperWick >= body * wickRejectMult

    strictEngulf =
         bear and
         pc > po and
         currBodyHigh >= prevBodyHigh and
         currBodyLow <= prevBodyLow

    flexEngulf =
         bear and
         c < prevBodyLow

    engulf =
         touchArea and
         closeOK and
         (allowFlexibleEngulfing ? (strictEngulf or flexEngulf) : strictEngulf)

    confirmMode == "Rejection Only" ? rejection :
     confirmMode == "Engulfing Only" ? engulf :
     rejection or engulf

        float l = array.get(m5LowArr, i)
        float c = array.get(m5CloseArr, i)

        float po = array.get(m5OpenArr, i - 1)
        float phh = array.get(m5HighArr, i - 1)
        float pll = array.get(m5LowArr, i - 1)
        float pc = array.get(m5CloseArr, i - 1)

        bool m5AfterRBS = rbsActive and not na(activeRBSReadyTime) and t >= activeRBSReadyTime
        bool m5AfterSBR = sbrActive and not na(activeSBRReadyTime) and t >= activeSBRReadyTime

        // line fail based on M5 close

            if c < activeRBSLine
                activeRBSLineFailed := true

        if m5AfterSBR and not activeSBRLineFailed and not na(activeSBRLine)

             m5AfterRBS and
             htfBuyOK and
             showBuyLineSignal and

             not activeRBSLineFailed and

             (na(lastBuyLineM5Time) or t > lastBuyLineM5Time)

            buyLineEntry := c
            buyLineTime := t
            activeRBSLineUsed := true
            lastBuyLineM5Time := t

             m5AfterRBS and
             htfBuyOK and
             showBuyDZSignal and

             not na(activeDemandTop) and
             not na(activeDemandBottom) and
             lineFailBuyOK and
             (na(lastBuyDZM5Time) or t > lastBuyDZM5Time)
