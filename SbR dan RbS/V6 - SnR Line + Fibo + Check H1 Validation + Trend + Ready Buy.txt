f_addLevel(float[] arr, float level, float tolerance, int maxLevels) =>
    bool exists = false
    if array.size(arr) > 0
        for i = 0 to array.size(arr) - 1
            if math.abs(array.get(arr, i) - level) <= tolerance
                exists := true
    if not exists
        array.unshift(arr, level)
        if array.size(arr) > maxLevels
            array.pop(arr)

f_nearestLevel(float[] arr, float level, float tolerance) =>
    float nearest = na
    float minDist = 999999999.0
    if array.size(arr) > 0
        for i = 0 to array.size(arr) - 1
            float lvl = array.get(arr, i)
            float dist = math.abs(lvl - level)
            if dist <= tolerance and dist < minDist
                minDist := dist
                nearest := lvl
    nearest

f_bullEngulf(float o, float h, float l, float c, float po, float ph, float pl, float pc) =>
    float prevBodyHigh = math.max(po, pc)
    float prevBodyLow  = math.min(po, pc)
    float currBodyHigh = math.max(o, c)
    float currBodyLow  = math.min(o, c)
    float crange = h - l
    float cbody  = math.abs(c - o)
    float cbodyPct = crange > 0 ? cbody / crange : 0.0

    bool prevBear = pc < po
    bool currBull = c > o

    bool strictBody =
         currBull and
         currBodyLow <= prevBodyLow and
         currBodyHigh >= prevBodyHigh

    bool closeBeyond =
         currBull and
         (
             c > ph or
             (allowCloseBeyondBodyOnly and c > prevBodyHigh)
         )

    bool strongReversal =
         currBull and
         cbodyPct >= minEngulfBodyPct and
         c > prevBodyHigh

    bool prevOK = not requirePrevOpposite or prevBear

    prevOK and
     (
         engulfMode == "Strict Body Engulf" ? strictBody :
         engulfMode == "Close Beyond Previous High/Low" ? closeBeyond :
         engulfMode == "Strong Reversal Candle" ? strongReversal :
         strictBody or closeBeyond or strongReversal
     )

f_bearEngulf(float o, float h, float l, float c, float po, float ph, float pl, float pc) =>
    float prevBodyHigh = math.max(po, pc)
    float prevBodyLow  = math.min(po, pc)
    float currBodyHigh = math.max(o, c)
    float currBodyLow  = math.min(o, c)
    float crange = h - l
    float cbody  = math.abs(c - o)
    float cbodyPct = crange > 0 ? cbody / crange : 0.0

    bool prevBull = pc > po
    bool currBear = c < o

    bool strictBody =
         currBear and
         currBodyHigh >= prevBodyHigh and
         currBodyLow <= prevBodyLow

    bool closeBeyond =
         currBear and
         (
             c < pl or
             (allowCloseBeyondBodyOnly and c < prevBodyLow)
         )

    bool strongReversal =
         currBear and
         cbodyPct >= minEngulfBodyPct and
         c < prevBodyLow

    bool prevOK = not requirePrevOpposite or prevBull

    prevOK and
     (
         engulfMode == "Strict Body Engulf" ? strictBody :
         engulfMode == "Close Beyond Previous High/Low" ? closeBeyond :
         engulfMode == "Strong Reversal Candle" ? strongReversal :
         strictBody or closeBeyond or strongReversal
     )
